import java.util.*;
class Solution {
    static int count;
    public int towerOfHanoi(int n, int from, int to, int aux) {
        
        count=0;
        hanoi(n,'a','b','c');
        return count;
    }
    void hanoi(int n,char a,char b,char c){
        if(n==0) return; 
        
        //n-1 disks from A to B via C
        hanoi(n-1,a,c,b);

        //largest from A to C
        ++count;

        //n-1 disks from B to C via A
        hanoi(n-1,b,a,c);
    }
}
