/* Structure of linked list Node
class Node
{
    int data;
    Node next;

    Node(int d) {
        data = d;
        next = null;
    }
}
*/
class Solution {
    int getKthFromLast(Node head, int k) {
        //slow=head, fast=head
        //make fast lead by k with a loop
        //in a loop, till fast!=null, make them take unity increment steps 
        //when f=null, we get s , which is kth node from end
        
        Node slow=head;
        Node fast=head;
        //make fast k steps ahead of slow
        for(int i=1;i<=k;i++){
            if(fast==null)return -1;
            fast=fast.next;
        }
        while(fast!=null){
            slow=slow.next;
            fast=fast.next;
        }
        return slow.data;
        
    }
}
