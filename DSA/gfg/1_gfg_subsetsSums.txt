class Solution {
    public ArrayList<Integer> subsetSums(int[] arr) {
        // code here
        ArrayList<Integer> sums=new ArrayList<>();
        helper(arr,0,0,sums);
        return sums;
    }
    public static void helper(int arr[],int sum,int idx,ArrayList<Integer> sums){
        if(idx==arr.length){  //after all branches
            sums.add(sum);
            return;
        }
        //take
        helper(arr,sum+arr[idx],idx+1,sums);
        
        //skip
        helper(arr,sum,idx+1,sums);
    }
}
