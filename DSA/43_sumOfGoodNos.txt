class Solution {
    public int sumOfGoodNumbers(int[] arr, int k) {
        int sum=0;
        for(int i=0; i<k;++i){
            if(arr[i]>arr[i+k]) sum+=arr[i];
            
        }
        for(int i=k;i<arr.length-k;++i){
            if(arr[i]>arr[i+k] && arr[i]>arr[i-k]){
                sum+=arr[i];
                }
        }
        for(int i=arr.length-k;i<arr.length;++i){
            if(arr[i]>arr[i-k]) sum+=arr[i];
        }
        return sum;
    }
}
