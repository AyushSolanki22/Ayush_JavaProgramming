//One approach  -- by natural thinking

class Solution {
    public int[] nodesBetweenCriticalPoints(ListNode head) {
        int ans[]=new int[2];
        int minDist=Integer.MAX_VALUE,maxDist=0;
        ArrayList <Integer> helper=new ArrayList<>();  //arraylist for storing all indexes of critical pts.
        int idx=0;    
        ListNode curr=head; ListNode prev=new ListNode(0); prev.next=curr;
        while(curr!=null){
            if(prev.val==0||curr.next==null){  //cannot include endmost nodes as critical points
                curr=curr.next;
                prev=prev.next;
            }
            else{
                if(curr.val>prev.val && curr.val>curr.next.val){  //critical pt
                    helper.add(idx);
                    idx++;
                    curr=curr.next;
                    prev=prev.next;
                }
                else if(curr.val<prev.val && curr.val<curr.next.val){  //critical pt.
                    helper.add(idx);
                    idx++;
                    curr=curr.next;
                    prev=prev.next;

                }
                else{         //normal node
                    idx++;
                    curr=curr.next;
                    prev=prev.next;
                }
            }

        }
        if(helper.size()>=2){  //to account for edge case -- only calculate minDist, maxDist when 2 or more critical pts
            maxDist=helper.get(helper.size()-1)-helper.get(0);
            for(int i=helper.size()-1; i>=1; --i){
                minDist=Math.min(helper.get(i)-helper.get(i-1), minDist);
            }
            ans[0]=minDist; ans[1]=maxDist;  
        }
        else{
            ans[0]=-1; ans[1]=-1;
        }
        return ans;
    }
}

class Solution {
    //using 3+2 variables -- idx, fidx, lidx, + maxDist, minDist
    //using 3 nodes
    public int[] nodesBetweenCriticalPoints(ListNode head) {

        int idx=1; int fidx=-1; int lidx=-1;   //idx=1 initially as b on 1st index
        int minDist=Integer.MAX_VALUE;
        int maxDist=-1;

        ListNode a=head; ListNode b=head.next; ListNode c=head.next.next;
        while(c!=null){
            if(b.val>a.val && b.val>c.val){
                if(fidx==-1) {
                    fidx=idx; lidx=idx; 
                }  
                else{     //for 2 or more critical points
                    int dist=idx-lidx;
                    minDist=Math.min(dist, minDist);     //dist=idx-lidx, for minDist calculation
                    lidx=idx;
                }
                idx++;
                a=a.next; b=b.next; c=c.next; 

            }
            else if(b.val<a.val && b.val<c.val){
                if(fidx==-1) {
                    fidx=idx; lidx=idx; 
                }
                else{     //for 2 or more critical points
                    int dist=idx-lidx;
                    minDist=Math.min(dist, minDist);
                    lidx=idx;
                    
                }
                idx++;
                a=a.next; b=b.next; c=c.next; 
            }
            else{
                idx++;
                a=a.next; b=b.next; c=c.next; 
            }
        }

        maxDist=lidx-fidx;

        if(fidx==lidx) return new int[]{-1,-1};   //considering for less than 2 critical pts
        else return new int[]{minDist, maxDist};
        
    }
}
