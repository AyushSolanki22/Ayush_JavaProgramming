import java.util.*;
class Solution {
    public int maximumDifference(int[] nums) {
        int n=nums.length;
        int maxDiff=0;
        for (int i=0; i<n; ++i) {
            for (int j=n-1; j>=n-2; --j) {
                int currDiff=nums[j]-nums[i];
                maxDiff=currDiff>maxDiff? currDiff:maxDiff;

            }

        }
        if (maxDiff>0) return maxDiff;
        else return -1;
    }
}
