class Solution {
    public int removeDuplicates(int[] nums) {
        //the array is already sorted , so duplicates will be adjacent.
        int i=0; //pointer
        for (int j=1; j<nums.length; ++j) {
            if (nums[j]!=nums[i]) {
                ++i;
                nums[i]=nums[j];}
                
            }
        return i+1; //the no. of unique elements
        }
     
}
