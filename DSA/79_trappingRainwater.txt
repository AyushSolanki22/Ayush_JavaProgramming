//One approach  -- using Auxiliary Arrays
//T.C= 0(n),, S.C= 0(n)
class Solution {
    public int trap(int[] height) {
        int n=height.length;

        //creating left max boundary -- array
        int leftMax[]=new int[n];
        leftMax[0]=height[0];
        for(int i=1;i<n;++i){
            leftMax[i]=Math.max(height[i],leftMax[i-1]);
        }

        //creating right max boundary -- array
        int rightMax[]=new int[n];
        rightMax[n-1]=height[n-1];
        for (int i=n-2; i>=0; --i) {
            rightMax[i]=Math.max(height[i], rightMax[i+1]);
        }

        int tr=0;
        //loop
        for (int i=0; i<n; ++i) {
            int w=Math.min(leftMax[i],rightMax[i]);   //waterlevel = w=  min(right max,left max);
            tr+=w-height[i];                   ////trapped rainwater= (water level-building height)*width 
        }
        
        return tr;

    }
}



//Another approach -- using 2 pointes

// The key is realizing that we don't need to store the entire arrays. for leftMax & rightMax
// we can use 2 pointer approach ,,,, T.C=0(n), now S.C becomes only 0(1).
class Solution {
    public int trap(int[] height) {

        int left = 0;
        int right = height.length - 1;

        int leftMax = 0;
        int rightMax = 0;

        int tr = 0;

        while(left < right){
            leftMax = Math.max(leftMax, height[left]);
            rightMax = Math.max(rightMax, height[right]);
            
            //concept--- The side having the smaller maximum boundary is safe to process because its water level is already fixed.
            if(leftMax <= rightMax){
                tr += leftMax - height[left];
                left++;
            }
            else{
                tr += rightMax - height[right];
                right--;
            }
        }

        return tr;
    }
}
