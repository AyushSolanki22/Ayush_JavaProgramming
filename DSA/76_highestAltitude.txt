class Solution {
    public int largestAltitude(int[] gain) {
        int max=0;
        int diff=0;   //gain will be higher when difference will be max overall at a point.
        for(int i=0; i<gain.length;++i){
            diff=diff+gain[i];
            max=diff>max?diff:max;
        }
        return max;
    }
}
