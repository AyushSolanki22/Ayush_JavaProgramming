//reverse linkedlist in between question


//One approach -- using arrayList  //imp- to make connections between nodes in arraylist after reversing
class Solution {
    public ListNode reverseBetween(ListNode head,int left, int right) {
        
        ArrayList<ListNode> arr = new ArrayList<>();
        ListNode temp = head;

        while(temp != null){
            arr.add(temp);
            temp = temp.next;
        }

        int i = left - 1;
        int j = right- 1;

        while(i < j){

            ListNode t1 = arr.get(i);
            ListNode t2 = arr.get(j);

            arr.set(i, t2);
            arr.set(j, t1);

            i++;
            j--;
        }

        for(int k = 0; k < arr.size(); k++){

            arr.get(k).next =(k == arr.size()-1)? null: arr.get(k+1);
        }

        return arr.get(0);
    }
}

 
//another approach
class Solution {
    public ListNode reverseBetween(ListNode head, int left, int right) {
        //concept - divided linkedList into 3 parts, 1 for reversing, 2 for maintaing connection  
        //head1 (dummy.next), tail1
        //head2, tail2     ,,,,,,  head3 , tail3

        ListNode dummy=new ListNode(0); ListNode temp=dummy; dummy.next=head;   
        //dummy node for a case if left=1

        ListNode tail1=temp;
        for(int i=1;i<left;++i){   
            tail1=tail1.next;
        }
        ListNode head2=tail1.next;

        ListNode tail2=temp;
        for(int i=1;i<=right;++i){   
            tail2=tail2.next;
        }
        ListNode head3=tail2.next;

        tail1.next=null; tail2.next=null;  ///to break connections

        //reverse
        reverse(head2);

        tail1.next=tail2;   //after reverse 
        head2.next=head3;
        return dummy.next; 
    }
    public static void reverse(ListNode head){
        ListNode prev=null;
        ListNode curr=head;
        ListNode fwd=null;
        while(curr!=null){
            fwd=curr.next;
            curr.next=prev;
            prev=curr;
            curr=fwd;
        }
        //no need to return, linkedlist reversed, just work done
    }
}
