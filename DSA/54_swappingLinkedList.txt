class Solution {
    public ListNode swapNodes(ListNode head, int k) {
        
        ListNode slow=head;
        ListNode fast=head;
        
        //make fast lead by k 
        for(int i=1;i<=k-1;++i){ //k-1 for 1-indexing
          fast=fast.next;
        }
        ListNode temp=fast; //for swapping
        while(fast.next!=null){
          slow=slow.next;
          fast=fast.next;
        }
        int temporary=slow.val;
        slow.val=temp.val;
        temp.val=temporary;
        return head;
    }
}
