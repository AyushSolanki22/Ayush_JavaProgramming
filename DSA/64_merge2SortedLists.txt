//One solution -- by natural thinking
class Solution {
    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        //create two nodes-i,j, one dummy node-k, traverse & merge
        
        ListNode i=list1;
        ListNode j=list2;
        ListNode k=new ListNode(0);  //this node is helping for merging linkedList 
        while(i!=null && j!=null){
            if(i.val<=j.val){
                k.next=i;
                k=i;
                i=i.next;
            }
            else{
                k.next=j;
                k=j;
                j=j.next;
            }
        }
        while(i!=null){   //for adding left nodes to the merged list
            k.next=i;
            k=i;
            i=i.next;
        }
        while(j!=null){
            k.next=j;
            k=j;
            j=j.next;
        }
        if(list1==null) return list2;   //null check - edge case 1
        else if(list2==null) return list1;

        if(list1.val<=list2.val) return list1;  //return check - edge case 2
        else return list2;    
    }
}


//Another solution -- optimised approach 
//1. store dummy, direct return dummy.next
//2. if one reaches null, don't have to traverse through another list fully

class Solution {
    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        //create two nodes-i,j, one dummy node-k, traverse & merge
        
        ListNode i=list1;
        ListNode j=list2;
        ListNode dummy=new ListNode(0);  
        ListNode k=dummy;      //this node is helping for merging linkedList 
        while(i!=null && j!=null){
            if(i.val<=j.val){
                k.next=i;
                k=i;
                i=i.next;
            }
            else{
                k.next=j;
                k=j;
                j=j.next;
            }
        }
        if(i==null) k.next=j;   //when one reaches null case
        else k.next=i;
        
        return dummy.next;  
    }
}
