class Solution {

    public int compress(char[] s) {

        int i = 0;
        StringBuilder sb = new StringBuilder();

        while(i < s.length){

            int j = i;

            // move j while same chars      //Sliding window concept
            while(j < s.length && s[i] == s[j]){
                j++;
            }

            // append char
            sb.append(s[i]);

            // append count if >1
            if(j - i > 1){
                sb.append(j - i);
            }

            // move to next group
            i = j;
        }

        // copy back into original array
        for(int k = 0; k < sb.length(); k++){
            s[k] = sb.charAt(k);
        }

        return sb.length();  //as the chars array is modified partially only till sb.length only , after that it contains original array.
    }
}
