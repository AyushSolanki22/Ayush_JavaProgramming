//One solution- by natural thinking
//find duplicate element & delete immediately
class Solution {
    Node removeDuplicates(Node head) {
        Node temp=head;
        Node prev=head;
        while(temp.next!=null){
            temp=temp.next;
            if(temp.data==prev.data){
                prev.next=temp.next;
            }
            else{
                prev=temp;
            }
        }
        return head;
    }
}



//Another Solution
// two pointer/runner technique 
class Solution {
    Node removeDuplicates(Node head) {
        Node i=head;
        Node j=head;
        while(j!=null){
            if(i.data==j.data){
                j=j.next;
            }
            else{
                i.next=j;
                i=j;
            }
        }
        i.next=j;
        return head;
    }
}
