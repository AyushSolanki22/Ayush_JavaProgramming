
//Clone list Next and Random
//Copy List with random pointer -- deep copy 


//HARDEST QUESTION
//step 1 -- creating deep copy without random connections   (of like simple ll) -- with help of dummy node 
//step 2 -- alternatively connect both LL  dummy node , (h1, h2) & t1,t2
//step 3 -- assigning random pointers - get t1,t2 to h1,h2---in loop, t2.random=t1.random.next; t1,t2=t1,t2.next.next ,,,random=null condition -- magic of because of alternate conncetions
//step 4 -- splitting the combined list  -- create d1, d2 (dummy nodes), t1, t2 & t (for traversing combined list)
class Solution {
    public Node copyRandomList(Node head1) {
        //step-1 : Creating deep copy without random
        Node head2=deepCopy(head1);

        //step-2 : Alternatively connecting lists  --- interweave concept 
        merge(head1,head2);

        //step 3 : Assigning random pointers
        randomConnections(head1, head2);

        //step 4 : Split the list
        split(head1);

        return head2;
    }

    public Node deepCopy(Node head1){
        Node temp1=head1;
        Node head2=new Node(-1);  //dummy for head2
        Node ans=head2;
        while(temp1!=null){
            Node t=new Node(temp1.val);
            ans.next=t;

            ans=ans.next;
            temp1=temp1.next;
        }
        ans.next=null;
        return head2.next;
    }

    public void merge(Node head1, Node head2){
        Node t1=head1, t2=head2;
        Node dummy=new Node(-1);
        Node t=dummy;
        while(t1!=null){
            t.next=t1;  t1=t1.next;
            t=t.next;
            t.next=t2;  t2=t2.next;
            t=t.next;
        }
    }

    public void randomConnections(Node head1, Node head2){
        Node t1=head1;
        Node t2=head2;
        while(t1!=null){
            if(t1.random==null) t2.random=null;
            else t2.random=t1.random.next;
            t1=t1.next.next;
            if(t2.next!=null) t2=t2.next.next;   //if condition -- edge case
        }
    }

    public void split(Node head1){
        Node dummy1=new Node(-1);
        Node t1=dummy1;
        Node dummy2=new Node(-1);
        Node t2=dummy2;
        Node temp=head1;  //for traversing merged list
        
        while(temp!=null){
            t1.next=temp;  t1=t1.next;
            temp=temp.next;

            t2.next=temp;  t2=t2.next;
            temp=temp.next;
        }
        t1.next=null;   //to break connection at last too of merged list
    }
}
