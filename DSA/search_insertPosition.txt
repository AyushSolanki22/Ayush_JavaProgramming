class Solution {
    public int searchInsert(int[] arr, int target) {
        int si =0;
        int ei=arr.length-1;
        // int mik=0;
        int mid=0;
        if(target>arr[ei] || target<arr[si]){         //Boundary Check (out of range check)
            if(target>arr[ei]){
                return ei+1;
            }
            return si;
        }
        while(si<=ei){
            mid=si+(ei-si)/2;
            if(arr[mid]==target){             //Base condition
                return mid;
            }
            else if(target<arr[mid] && target>arr[mid-1]){         (eek auur just eek phle ki condition incase mid , si ke bilkul nazdeek ho )
                return mid;
            }
            else if(target<arr[mid]){                    (normal condition -- eek se phle walo ke liye )
                ei=mid-1;
            }
            else{
                si=mid+1;                             (normal -- eek se aage walo ke liye)
            }

        }
            return -1;
    }
}
