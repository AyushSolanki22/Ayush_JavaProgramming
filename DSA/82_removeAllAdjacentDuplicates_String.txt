class Solution {
    public String removeDuplicates(String s) {
        Stack <Character> st=new Stack<>();
        StringBuilder ans=new StringBuilder();

        //to remove all adjacent duplicates using stack
        for(int i=0; i<s.length();++i){
            char ch=s.charAt(i);
            if(st.size()==0) {
                st.push(ch);
            }
            else{
                char top=st.peek();

                if(top==ch) st.pop();
                else {
                    st.push(ch);
                }
            }
        }

        //traverse stack and store in new string 
        while(st.size()>0){
            ans.append(st.pop());
        }

        //and reverse new string to get desired ans
        ans.reverse();

        return ans.toString();


    }
}
