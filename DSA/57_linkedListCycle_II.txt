//question - to return first node of a cycle/loop in LinkedList

public class Solution {
    public ListNode detectCycle(ListNode head) {
        //A type of linkedlist in which last node is not connected to null, it is connected to some other node in the linkedList itself , creating a cycle in a LinkedList (a loop in a linkedList)
        
        //10-->20-->30-->40-->50-->60 ,, 60-->30 (a repeated cycle,30-->40-->50-->60-->30-->.....)
        
        //Concept: minute hand & second hand meets each other at a while, moving at significant difference in speeds
        //So, Approach: slow fast approach
        ListNode fast=head;
        ListNode slow=head;
        ListNode b=head;
        
        boolean isCycle=false;
        while(fast!=null && fast.next!=null){ //considering odd & even length case both
            fast=fast.next.next;
            slow=slow.next;
            if(fast==slow){
              isCycle=true;
              break;
            }
        }
        //By proof of first node derived from proof of detecting loop in linkedlist, we can see that b,slow taking unity steps (after slow becoming equal to fast) , meets at first node.
        if(isCycle){
          while(b!=null){
          if(b==slow) return b;
          slow=slow.next;
          b=b.next;
         
        }
        }

        return null;
    }
}
