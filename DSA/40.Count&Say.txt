class Solution {
    public String countAndSay(int n) {
        if(n == 1){
            return "1";
        }

        String prev = countAndSay(n - 1);  //last string (as in this nth number is result od count and write digit of last one)

        StringBuilder ans = new StringBuilder();  //String for at current row

        int i = 0;

        while(i < prev.length()){

            int j = i;

            // count same digits
            while(j < prev.length() &&
                  prev.charAt(i) == prev.charAt(j)){

                j++;
            }

            int count = j - i;

            ans.append(count);
            ans.append(prev.charAt(i));

            i = j;
        }

        return ans.toString();
    }
}
